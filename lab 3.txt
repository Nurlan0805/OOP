
lst = [1, 2, 3, 4, 5]
print("Обычный список:", lst)
print("Обратный список:", lst[::-1]) 



def list_sort(lst):
    return sorted(lst, key=abs, reverse=True)

nums = [3, -7, 1, -10, 5]
print("Исходный:", nums)
print("Отсортированный:", list_sort(nums))  



def change(lst):
    lst[0], lst[-1] = lst[-1], lst[0]
    return lst

example = [10, 20, 30, 40]
print("До:", example)
print("После:", change(example))
